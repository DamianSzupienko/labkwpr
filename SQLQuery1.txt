USE master
DROP DATABASE firma_komputerowa
CREATE DATABASE firma_komputerowa
USE firma_komputerowa

create table Produkty 
(
model varchar(30) PRIMARY KEY NOT NULL,
producent varchar(30),
typ varchar(10),
);

create table PC
(model varchar(30) PRIMARY KEY NOT NULL,
procesor int,
ram int,
hdd int,
cd int,
ekran int,
cena money,
);

create table Laptop
(model varchar(30) PRIMARY KEY NOT NULL,
procesor int,
ram int,
hdd int,
ekran int,
cena money,
);

create table Drukarki
(model varchar(30) PRIMARY KEY NOT NULL,
kolor bit,
typ varchar(20),
cena money,
);

ALTER TABLE PC
ADD CONSTRAINT fk_Produkty_PCFOREIGN KEY (model)REFERENCES Produkty(model)

ALTER TABLE Laptop
ADD CONSTRAINT fk_Produkty_LaptopFOREIGN KEY (model)REFERENCES Produkty(model)

ALTER TABLE Drukarki
ADD CONSTRAINT fk_Produkty_DrukarkiFOREIGN KEY (model)REFERENCES Produkty(model)
